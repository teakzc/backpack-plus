local TS = require(game:GetService("ReplicatedStorage"):WaitForChild("rbxts_include"):WaitForChild("RuntimeLib"))
local _react = TS.import(script, game:GetService("ReplicatedStorage"), "rbxts_include", "node_modules", "@rbxts", "react")
local React = _react
local forwardRef = _react.forwardRef
local Frame = TS.import(script, game:GetService("ReplicatedStorage"), "Package", "client", "ui", "core", "frame").Frame
local usePx = TS.import(script, game:GetService("ReplicatedStorage"), "Package", "client", "ui", "hooks", "usePx").usePx
--[[
	*
	 * Creates a scrolling frame
	 *
	 * @hidden
	 
]]
local Scrolling = forwardRef(function(props, ref)
	local px = usePx()
	local _attributes = {
		ref = ref,
		Size = props.size or UDim2.fromScale(1, 1),
		Position = props.position or UDim2.fromScale(0.5, 0.5),
		AnchorPoint = props.anchorPoint or Vector2.new(0.5, 0.5),
		Rotation = props.rotation,
		BackgroundColor3 = props.backgroundColor,
	}
	local _condition = props.backgroundTransparency
	if _condition == nil then
		_condition = 1
	end
	_attributes.BackgroundTransparency = _condition
	_attributes.ClipsDescendants = props.clipsDescendants
	_attributes.Visible = props.visible
	_attributes.ZIndex = props.zIndex
	_attributes.LayoutOrder = props.layoutOrder
	_attributes.BorderSizePixel = 0
	_attributes.Event = props.event
	_attributes.Change = props.change
	_attributes.AutomaticCanvasSize = props.automaticCanvasSize
	_attributes.BottomImage = props.bottomImage
	_attributes.MidImage = props.midImage
	_attributes.ScrollBarImageColor3 = props.scrollBarImageColor3
	_attributes.ScrollBarImageTransparency = props.scrollBarImageTransparency
	_attributes.ScrollBarThickness = props.scrollBarThickness
	_attributes.ScrollingDirection = props.scrollingDirection
	_attributes.ScrollingEnabled = props.scrollingEnabled
	_attributes.TopImage = props.topImage
	_attributes.VerticalScrollBarInset = props.verticalScrollBarInSet
	_attributes.VerticalScrollBarPosition = props.verticalScrollBarPosition
	_attributes.CanvasPosition = props.canvasPosition
	_attributes.CanvasSize = props.canvasSize
	_attributes.ElasticBehavior = props.elasticBehavior
	_attributes.HorizontalScrollBarInset = props.horizontalScrollBarInSet
	return React.createElement("scrollingframe", _attributes, if props.disableInset then (props.children) else (React.createElement(Frame, {
		backgroundTransparency = 1,
		size = UDim2.new(1, px(-35), 1, px(-35)),
	}, props.children)))
end)
return {
	Scrolling = Scrolling,
}
